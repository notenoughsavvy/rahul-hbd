<body>
  <h1>Are you brave enough?</h1>
  <button id="scareButton">Click Me</button>

  <video id="jumpscareVideo" src="https://example.com/scary-video.mp4" preload="auto"></video>

  <script>
    const scareButton = document.getElementById('scareButton');
    const jumpscareVideo = document.getElementById('jumpscareVideo');
    const body = document.body;

    scareButton.addEventListener('click', () => {
      // Flash background
      body.classList.add('flash');

      // Show and play video
      jumpscareVideo.style.display = 'block';
      jumpscareVideo.play();

      // Fullscreen request (optional)
      if (jumpscareVideo.requestFullscreen) {
        jumpscareVideo.requestFullscreen();
      } else if (jumpscareVideo.webkitRequestFullscreen) {
        jumpscareVideo.webkitRequestFullscreen();
      }

      // Remove button and heading
      scareButton.remove();
      document.querySelector('h1').remove();

      // Stop flashing after 5s
      setTimeout(() => {
        body.classList.remove('flash');
      }, 5000);
    });
  </script>
